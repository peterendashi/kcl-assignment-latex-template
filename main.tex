    \stepcounter{questioncounter}
    \setcounter{partcounter}{0}
    \vspace{1em}
    {\sffamily\bfseries Question \thequestioncounter
    \ifx&#1&\else: #1\fi}
    \par\vspace{0.3em}
}

    \setcounter{partcounter}{0}
}{}

\renewcommand{\part}[1]{
    \stepcounter{partcounter}
    \vspace{0.7em}
    {\sffamily\bfseries (\alph{partcounter}) #1}
    \par\vspace{0.3em}
}
